int main() {
	int i = 10;
	int *pi = &amp;i; //이는 int *p; p=&amp;i와 같다. *p=&amp;i가 아니다
	printf("i = %d, pi = %u\n", i, *pi);
	(*pi)++; //pi의 주소를 참조한 값은 1 올린다
	printf("i = %d, *pi = %u\n\n", i, *pi);
	*pi++; //참조하기전 pi의 값을 올린다
	printf("i = %d, *pi = %u\n\n", i, *pi); //*pi보다 4 큰 메모리의 주소에 있는 데이터이다
}</div>
