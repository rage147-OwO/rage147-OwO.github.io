int main() {
	FILE *fp = NULL;

	fp = fopen("test.txt", "w"); //쓰기 모드로 연다
	int a[5];
	for (int i = 0; i &lt; 5; i++)
	{
		scanf("%d", &amp;a[i]); //a[i]에 입력받는다
		fprintf(fp, "%d\n", a[i]); //a[i]\n을 파일에 쓴다
	}
	fclose(fp);


	fp = fopen("test.txt", "r");
	int b[5];
	for (int i = 0; i &lt; 5; i++)
	{
		fscanf(fp,"%d", &amp;b[i]); //b[i]를 입력받는다
		printf("%d\n",b[i]); //b[i]를 출력한다
	}
	fclose(fp);
}</div>
