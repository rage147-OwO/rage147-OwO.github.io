int main() {
	int x, y, q;
	scanf("%d%d", &amp;x, &amp;y);
	if ((x == y) &amp;&amp; x == 0)
		puts("x,y는 0입니다");
	else {
		if (x*y &gt;= 0)
			q = 1;
		else
			q = 2;
		(y &gt;= 0) ? (q = q) : (q = q + 2);
		printf("%d분면 입니다,", q);
	}
	return 0;
}</div>
