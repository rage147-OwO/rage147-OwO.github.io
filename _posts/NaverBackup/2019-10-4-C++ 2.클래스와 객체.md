<div class="__se_code_view language-javascript">#include "pch.h"
#include &lt;iostream&gt;
using namespace std;
class circle {
public:
	int radius;
	circle();
	circle(int r);
	double getarea();
};
circle::circle() {
	radius = 1;
}
circle::circle(int r) {
	radius = r;
}
double circle::getarea() {
	return radius * radius*3.14;
}
int main()
{
	circle a;
	cout &lt;&lt; a.getarea() &lt;&lt; endl;

	circle b(10);
	cout &lt;&lt; b.getarea() &lt;&lt; endl;
}</div>
