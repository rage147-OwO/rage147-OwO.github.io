int main(void) {
	int x = 0, y = 0, result;
	
    result = 2 &gt; 3 || 6 &gt; 7; //관계연산자가 논리연산자보다 우선순의가 높다
	printf("%d\n", result);
	
	result = 2 &gt; 3 || 3 &gt; 2; 
	printf("%d\n", result);

	result = x = y = 1; //대입연산자는 오른쪽에서 왼쪽이다
	printf("%d\n", result);
	
	result = - ++x + y--; 
//-부호와 ++연산자의 우선순위는 같다 따라서 결합 규칙에 따라 오른쪽의 ++계산 후 -부호가 붙는다
	printf("%d\n", result);

}</div>
