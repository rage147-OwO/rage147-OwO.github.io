void print_stars() { //아무것도 입력받지 않는다
	for (int i = 0; i &lt; 30; i++) // *을 30번 출력한다
		printf("*");
}
int main(void) {
	print_stars(); //함수 실행
	printf("\nHello World!\n");
	print_stars(); //함수 실행
	printf("\n");
	for (size_t i = 0; i &lt; 100; i++) 
	{
		print_stars(); //함수를 100번 반복해 실행
		printf("\n");
	}
	return 0;

}</div>
