int main() {
	int a[3];
	a[0] = 1;
	a[1] = 2;
	a[2] = 3;
	
	int *ap = &amp;a[0]; //a[0]의 주소를 포인터변수 ap에 저장
	
	printf("%u %u %u\n", a[0], a[1], a[2]);
	printf("%u %u %u\n", *(ap), *(ap + 1), *(ap + 2)); //ap, ap+1 , ap+2
}</div>
